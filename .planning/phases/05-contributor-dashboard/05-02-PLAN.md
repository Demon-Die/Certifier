// src/lib/realtime.ts
import { createBrowserClient } from '@/lib/supabase/client';
import { RealtimeChannel } from '@supabase/supabase-js';

export function createProfileChannel(userId: string, callback: (payload: any) => void): RealtimeChannel {
  const supabase = createBrowserClient();
  return supabase
    .channel(`profile:${userId}`)
    .on(
      'postgres_changes',
      { event: 'UPDATE', schema: 'public', table: 'profiles', filter: `id=eq.${userId}` },
      callback
    )
    .subscribe();
}

export function createBadgeClaimsChannel(userId: string, callback: (payload: any) => void): RealtimeChannel {
  const supabase = createBrowserClient();
  return supabase
    .channel(`badge_claims:${userId}`)
    .on(
      'postgres_changes',
      { event: '*', schema: 'public', table: 'badge_claims', filter: `user_id=eq.${userId}` },
      callback
    )
    .subscribe();
}
```

// src/hooks/use-realtime-profile.ts
'use client';
import { useEffect, useState, useCallback } from 'react';
import { createBrowserClient } from '@/lib/supabase/client';
import { getFamilyProgress, type FamilyProgress } from '@/lib/points';
import type { DashboardData } from '@/lib/dashboard';

interface UseRealtimeProfileReturn {
  data: DashboardData | null;
  loading: boolean;
  error: Error | null;
}

export function useRealtimeProfile(userId: string): UseRealtimeProfileReturn {
  const [data, setData] = useState<DashboardData | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);
  
  // Initial fetch
  const fetchData = useCallback(async () => {
    try {
      const supabase = createBrowserClient();
      const { data: profile, error } = await supabase
        .from('profiles')
        .select('points_frontend, points_backend, points_docs, points_ideas, points_community')
        .eq('id', userId)
        .single();
      
      if (error) throw error;
      if (!profile) throw new Error('Profile not found');
      
      const families: Array<'frontend'|'backend'|'docs'|'ideas'|'community'> = 
        ['frontend', 'backend', 'docs', 'ideas', 'community'];
      const familyPointsMap = {
        frontend: profile.points_frontend,
        backend: profile.points_backend,
        docs: profile.points_docs,
        ideas: profile.points_ideas,
        community: profile.points_community,
      };
      
      const familiesProgress = families.map(family => getFamilyProgress(family, familyPointsMap[family]));
      const totalPoints = Object.values(familyPointsMap).reduce((sum, p) => sum + p, 0);
      
      setData({ families: familiesProgress, totalPoints });
    } catch (err) {
      setError(err as Error);
    } finally {
      setLoading(false);
    }
  }, [userId]);
  
  useEffect(() => {
    fetchData();
    
    // Subscribe to profile changes
    const supabase = createBrowserClient();
    const channel = supabase
      .channel(`profile:${userId}`)
      .on(
        'postgres_changes',
        { event: 'UPDATE', schema: 'public', table: 'profiles', filter: `id=eq.${userId}` },
        () => {
          fetchData(); // Refetch on change
        }
      )
      .subscribe();
    
    return () => {
      supabase.removeChannel(channel);
    };
  }, [userId, fetchData]);
  
  return { data, loading, error };
}
```
