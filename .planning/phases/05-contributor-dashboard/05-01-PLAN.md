```typescript
// src/lib/dashboard.ts
import { createServerClient } from '@/lib/supabase/server';
import { getProgressToNext, type Family } from '@/lib/points';

export interface FamilyProgress {
  family: Family;
  points: number;
  currentTier: string | null;
  nextTier: { tier: string; threshold: number } | null;
  progress: { current: number; target: number; percentage: number } | null;
}

export interface DashboardData {
  families: FamilyProgress[];
  totalPoints: number;
}

export async function getDashboardData(userId: string): Promise<DashboardData> {
  const supabase = createServerClient();
  
  const { data: profile, error } = await supabase
    .from('profiles')
    .select('points_frontend, points_backend, points_docs, points_ideas, points_community')
    .eq('id', userId)
    .single();
  
  if (error || !profile) {
    throw new Error('Profile not found');
  }
  
  const families: Family[] = ['frontend', 'backend', 'docs', 'ideas', 'community'];
  const familyPointsMap = {
    frontend: profile.points_frontend,
    backend: profile.points_backend,
    docs: profile.points_docs,
    ideas: profile.points_ideas,
    community: profile.points_community,
  };
  
  const familiesProgress = families.map(family => {
    const points = familyPointsMap[family];
    return getFamilyProgress(family, points);
  });
  
  const totalPoints = Object.values(familyPointsMap).reduce((sum, p) => sum + p, 0);
  
  return { families: familiesProgress, totalPoints };
}

export function getFamilyProgress(family: Family, points: number): FamilyProgress {
  const progress = getProgressToNext(points, family);
  const currentTier = progress ? (points >= 5 ? 'imp' : null) : 'demon king';
  // Actually calculate current tier properly
  let currentTierResult: string | null = null;
  if (points >= 135) currentTierResult = 'demon king';
  else if (points >= 45) currentTierResult = 'overlord';
  else if (points >= 15) currentTierResult = 'fiend';
  else if (points >= 5) currentTierResult = 'imp';
  
  return {
    family,
    points,
    currentTier: currentTierResult,
    nextTier: progress ? { tier: progress.tier, threshold: progress.threshold } : null,
    progress: progress ? { current: progress.current, target: progress.target, percentage: progress.percentage } : null,
  };
}
```
