// src/lib/points.ts

export const VALID_FAMILIES = ['frontend', 'backend', 'docs', 'ideas', 'community'] as const;
export type Family = typeof VALID_FAMILIES[number];

export const VALID_TIERS = ['imp', 'fiend', 'overlord', 'demon king'] as const;
export type Tier = typeof VALID_TIERS[number];

export const TIER_POINTS: Record<Tier, number> = {
  imp: 1,
  fiend: 3,
  overlord: 9,
  'demon king': 27,
};

export const BADGE_THRESHOLDS: Record<Tier, number> = {
  imp: 5,
  fiend: 15,
  overlord: 45,
  'demon king': 135,
};

export const BADGE_TIER_ORDER: Tier[] = ['imp', 'fiend', 'overlord', 'demon king'];

export function calculatePoints(tier: Tier): number {
  return TIER_POINTS[tier] || 0;
}

export function getBadgeTier(points: number, family: Family): Tier | null {
  // Returns highest tier achieved for given points in a family
  for (let i = BADGE_TIER_ORDER.length - 1; i >= 0; i--) {
    const tier = BADGE_TIER_ORDER[i];
    if (points >= BADGE_THRESHOLDS[tier]) return tier;
  }
  return null;
}

export function getNextBadgeThreshold(points: number, family: Family): { tier: Tier; threshold: number } | null {
  // Returns next tier and points needed
  const currentTier = getBadgeTier(points, family);
  if (!currentTier) return { tier: 'imp', threshold: BADGE_THRESHOLDS.imp };
  
  const currentIndex = BADGE_TIER_ORDER.indexOf(currentTier);
  if (currentIndex >= BADGE_TIER_ORDER.length - 1) return null; // Max tier
  
  const nextTier = BADGE_TIER_ORDER[currentIndex + 1];
  return { tier: nextTier, threshold: BADGE_THRESHOLDS[nextTier] };
}

export function getProgressToNext(points: number, family: Family): { current: number; target: number; percentage: number } | null {
  const next = getNextBadgeThreshold(points, family);
  if (!next) return { current: points, target: points, percentage: 100 };
  
  const currentTier = getBadgeTier(points, family);
  const currentThreshold = currentTier ? BADGE_THRESHOLDS[currentTier] : 0;
  const progress = points - currentThreshold;
  const needed = next.threshold - currentThreshold;
  
  return {
    current: progress,
    target: needed,
    percentage: Math.min(100, Math.round((progress / needed) * 100)),
  };
}

export function isValidFamily(family: string): family is Family {
  return VALID_FAMILIES.includes(family as Family);
}

export function isValidTier(tier: string): tier is Tier {
  return VALID_TIERS.includes(tier as Tier);
}

  family: family as Family,
  tier: tier as Tier,
  label,
  points: calculatePoints(tier as Tier),

import { describe, it, expect } from 'vitest';
import {
  TIER_POINTS,

  calculatePoints,
  getBadgeTier,
  getNextBadgeThreshold,
  getProgressToNext,
  isValidFamily,
  isValidTier,

describe('Points Calculation', () => {
  it('TIER_POINTS has correct values', () => {
    expect(TIER_POINTS.imp).toBe(1);
    expect(TIER_POINTS.fiend).toBe(3);
    expect(TIER_POINTS.overlord).toBe(9);
    expect(TIER_POINTS['demon king']).toBe(27);

  it('calculatePoints returns correct points', () => {
    expect(calculatePoints('imp')).toBe(1);
    expect(calculatePoints('fiend')).toBe(3);
    expect(calculatePoints('overlord')).toBe(9);
    expect(calculatePoints('demon king')).toBe(27);
    expect(calculatePoints('invalid')).toBe(0);

  it('getBadgeTier returns correct tier for points', () => {
