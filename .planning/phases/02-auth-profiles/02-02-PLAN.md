```typescript
// src/lib/rbac.ts
export type Role = 'contributor' | 'maintainer' | 'admin';

export const ROLE_HIERARCHY: Record<Role, number> = {
  contributor: 0,
  maintainer: 1,
  admin: 2,
};

export const ROUTE_PERMISSIONS: Record<string, Role> = {
  '/dashboard': 'contributor',
  '/maintainer': 'maintainer',  // future
  '/admin': 'admin',
  '/api/admin': 'admin',
  '/api/maintainer': 'maintainer',
};

export function hasRole(userRole: Role, requiredRole: Role): boolean {
  return ROLE_HIERARCHY[userRole] >= ROLE_HIERARCHY[requiredRole];
}

export function canAccess(userRole: Role, path: string): boolean {
  const required = ROUTE_PERMISSIONS[path];
  if (!required) return true; // Public route
  return hasRole(userRole, required);
}

export function getRequiredRole(path: string): Role | null {
  return ROUTE_PERMISSIONS[path] || null;
}

// Client-side hook
export function useRoleAccess(requiredRole: Role): boolean {
  const { data: session } = useSession();
  if (!session?.user?.role) return false;
  return hasRole(session.user.role, requiredRole);
}
```

Also create server-side version for middleware/API routes.
