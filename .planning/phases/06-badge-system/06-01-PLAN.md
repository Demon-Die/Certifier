```typescript
// src/lib/certifier.ts
const CERTIFIER_API_BASE = 'https://api.certifier.io/v1';

export interface CertifierCredentialRequest {
  recipientEmail: string;
  badgeTemplateId: string;
  metadata?: Record<string, string>;
}

export interface CertifierCredentialResponse {
  id: string;
  status: string;
  credential_url?: string;
  image_url?: string;
  embed_url?: string;
  expires_at?: string;
}

export class CertifierError extends Error {
  constructor(
    message: string,
    public statusCode: number,
    public isRateLimited: boolean = false
  ) {
    super(message);
    this.name = 'CertifierError';
  }
}

export async function claimCredential(
  request: CertifierCredentialRequest
): Promise<CertifierCredentialResponse> {
  const response = await fetch(`${CERTIFIER_API_BASE}/credentials`, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${process.env.CERTIFIER_API_KEY}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      recipientEmail: request.recipientEmail,
      badgeTemplateId: request.badgeTemplateId,
      metadata: request.metadata || {},
    }),
  });
  
  if (!response.ok) {
    if (response.status === 429) {
      throw new CertifierError(
        'Monthly quota exceeded. Badge claiming is temporarily unavailable.',
        response.status,
        true
      );
    }
    const errorBody = await response.text().catch(() => '');
    throw new CertifierError(
      `certifier.io API error: ${response.status} ${errorBody}`,
      response.status
    );
  }
  
  return response.json();
}

export async function getCredential(credentialId: string): Promise<CertifierCredentialResponse | null> {
  const response = await fetch(`${CERTIFIER_API_BASE}/credentials/${credentialId}`, {
    headers: {
      'Authorization': `Bearer ${process.env.CERTIFIER_API_KEY}`,
    },
  });
  if (!response.ok) return null;
  return response.json();
}
```

```typescript
// src/app/api/badges/claim/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { createServerClient } from '@/lib/supabase/server';
import { auth } from '@/lib/auth';
import { claimCredential, CertifierError } from '@/lib/certifier';
import { getTemplateId, getBadgeMeta } from '@/lib/badges';
import type { Family, Tier } from '@/lib/points';

export async function POST(request: NextRequest) {
  // 1. Authenticate
  const session = await auth();
  if (!session?.user?.id) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }
  
  // 2. Parse body
  const body = await request.json();
  const { family, tier, email } = body as { family: string; tier: string; email?: string };
  
  // 3. Validate family & tier
  const validFamilies = ['frontend', 'backend', 'docs', 'ideas', 'community'];
  const validTiers = ['imp', 'fiend', 'overlord', 'demon king'];
  if (!validFamilies.includes(family) || !validTiers.includes(tier)) {
    return NextResponse.json({ error: 'Invalid family or tier' }, { status: 400 });
  }
  
  const supabase = createServerClient();
  
  // 4. Verify badge claim exists and is available
  const { data: claim } = await supabase
    .from('badge_claims')
    .select('*')
    .eq('user_id', session.user.id)
    .eq('family', family)
    .eq('tier', tier)
    .eq('status', 'available')
    .single();
  
  if (!claim) {
    return NextResponse.json({ error: 'Badge not available for claim' }, { status: 404 });
  }
  
  // 5. Determine recipient email
  const recipientEmail = email || session.user.email || `${session.user.githubUsername}@users.noreply.github.com`;
  
  // 6. Get template ID
  const templateId = getTemplateId(family as Family, tier as Tier);
  if (!templateId) {
    return NextResponse.json({ error: 'Badge template not configured' }, { status: 500 });
  }
  
  // 7. Call certifier.io
  try {
    const result = await claimCredential({
      recipientEmail,
      badgeTemplateId: templateId,
      metadata: {
        github_username: session.user.githubUsername,
        family,
        tier,
      },
    });
    
    // 8. Update badge_claims
    const { error: updateError } = await supabase
      .from('badge_claims')
      .update({
        status: 'claimed',
        certifier_credential_id: result.id,
        claimed_at: new Date().toISOString(),
      })
      .eq('id', claim.id);
    
    if (updateError) {
      console.error('Failed to update badge_claims:', updateError);
      // Still return success since certifier issued it
    }
    
    const badgeMeta = getBadgeMeta(family as Family, tier as Tier);
    
    return NextResponse.json({
      success: true,
      credentialId: result.id,
      credentialUrl: result.credential_url,
      imageUrl: result.image_url || result.embed_url,
      badge: badgeMeta,
    });
    
  } catch (error) {
    if (error instanceof CertifierError) {
      if (error.isRateLimited) {
        return NextResponse.json({ error: error.message, rateLimited: true }, { status: 429 });
      }
      return NextResponse.json({ error: error.message }, { status: error.statusCode || 500 });
    }
    console.error('Unexpected certifier error:', error);
    return NextResponse.json({ error: 'Badge claiming failed. Try again later.' }, { status: 500 });
  }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → Claim API | Authenticated user claims badge |
| Claim API → certifier.io | Server-side API key, not exposed to client |
| Claim API → Supabase | Service role updates badge_claims |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-01 | Spoofing | Claim another user's badge | mitigate | API validates badge claim ownership via auth user_id |
| T-06-02 | Tampering | Claim same badge twice | mitigate | Unique constraint on (user_id, family, tier), status check before claim |
| T-06-03 | Info Disclosure | certifier API key leak | mitigate | API key server-only (process.env), never sent to client |
| T-06-04 | DoS | certifier.io quota exceeded | mitigate | Catch 429, return friendly message, allow retry next month |
| T-06-05 | Spoofing | Wrong template ID | accept | Configured via env vars, validated at build time |
</threat_model>
