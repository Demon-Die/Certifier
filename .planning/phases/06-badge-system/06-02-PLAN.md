Update BadgeClaimCard from Phase 05 with comprehensive error handling:

```tsx
// src/components/dashboard/badge-claim-card.tsx - update claim handler
const handleClaim = async () => {
  setClaiming(true);
  setError(null);
  
  try {
    const res = await fetch('/api/badges/claim', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ family, tier }),
    });
    
    const data = await res.json();
    
    if (!res.ok) {
      if (data.rateLimited) {
        // Rate limit - show friendly message
        toast({
          title: 'Monthly Quota Exceeded',
          description: 'Badge claiming is temporarily unavailable. The quota resets at the start of next month.',
          variant: 'destructive',
        });
      } else if (res.status === 429) {
        toast({
          title: 'Too Many Requests',
          description: 'Please wait a moment before trying again.',
          variant: 'destructive',
        });
      } else {
        throw new Error(data.error || 'Claim failed');
      }
      return;
    }
    
    // Success
    toast({
      title: 'Badge Claimed! 🎉',
      description: `Your ${data.badge.name} badge has been issued.`,
    });
    
    setClaimed(true);
    setImageUrl(data.imageUrl);
    setCredentialUrl(data.credentialUrl);
    
    // Notify parent to refetch claimable badges
    onClaimSuccess?.();
    
  } catch (err) {
    toast({
      title: 'Claim Failed',
      description: err instanceof Error ? err.message : 'An unexpected error occurred. Please try again.',
      variant: 'destructive',
    });
  } finally {
    setClaiming(false);
  }
};
```

3. **Claimed**: Green checkmark, "Claimed!", show credential link
