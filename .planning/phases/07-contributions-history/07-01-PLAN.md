```typescript
// src/lib/contributions.ts
import { createServerClient } from '@/lib/supabase/server';
import { PostgrestFilterBuilder } from '@supabase/postgrest-js';

export interface Contribution {
  id: string;
  user_id: string;
  repo: string;
  pr_number: number;
  pr_title: string;
  pr_url: string;
  merged_at: string;
  family: string;
  tier: string;
  points_awarded: number;
  label_used: string;
  created_at: string;
}

export interface ContributionsQuery {
  userId?: string;
  family?: string;
  tier?: string;
  repo?: string;
  dateFrom?: string;
  dateTo?: string;
  page?: number;
  pageSize?: number;
}

export interface PaginatedResult<T> {
  data: T[];
  total: number;
  page: number;
  pageSize: number;
  totalPages: number;
}

export async function getContributions(
  query: ContributionsQuery
): Promise<PaginatedResult<Contribution>> {
  const supabase = createServerClient();
  const page = query.page || 1;
  const pageSize = query.pageSize || 20;
  const offset = (page - 1) * pageSize;
  
  // Build query
  let sql = supabase
    .from('contributions')
    .select('*', { count: 'exact' });
  
  if (query.userId) {
    sql = sql.eq('user_id', query.userId);
  }
  if (query.family) {
    sql = sql.eq('family', query.family);
  }
  if (query.tier) {
    sql = sql.eq('tier', query.tier);
  }
  if (query.repo) {
    sql = sql.ilike('repo', `%${query.repo}%`);
  }
  if (query.dateFrom) {
    sql = sql.gte('merged_at', query.dateFrom);
  }
  if (query.dateTo) {
    sql = sql.lte('merged_at', query.dateTo);
  }
  
  // Order by most recent first
  sql = sql.order('merged_at', { ascending: false });
  
  // Paginate
  sql = sql.range(offset, offset + pageSize - 1);
  
  const { data, error, count } = await sql;
  
  if (error) {
    console.error('Failed to fetch contributions:', error);
    throw new Error('Failed to fetch contributions');
  }
  
  return {
    data: (data as Contribution[]) || [],
    total: count || 0,
    page,
    pageSize,
    totalPages: Math.ceil((count || 0) / pageSize),
  };
}

export async function getUserContributions(
  userId: string,
  query: Omit<ContributionsQuery, 'userId'> = {}
): Promise<PaginatedResult<Contribution>> {
  return getContributions({ ...query, userId });
}
```

```tsx
export default async function ContributionsPage({
  searchParams,
}: {
  searchParams: { family?: string; tier?: string; repo?: string; dateFrom?: string; dateTo?: string; page?: string };
}) {
  const session = await auth();
  if (!session?.user?.id) redirect('/auth/signin');
  
  const page = parseInt(searchParams.page || '1');
  const initialData = await getUserContributions(session.user.id, {
    family: searchParams.family,
    tier: searchParams.tier,
    repo: searchParams.repo,
    dateFrom: searchParams.dateFrom,
    dateTo: searchParams.dateTo,
    page,
    pageSize: 20,
  });
  
  return (
    <div className="container mx-auto py-8">
      <h1 className="text-3xl font-bold mb-2">Contributions History</h1>
      <p className="text-muted-foreground mb-6">All your merged PR contributions across all tracked repositories</p>
      <ContributionsTable initialData={initialData} userId={session.user.id} />
    </div>
  );
}
```
