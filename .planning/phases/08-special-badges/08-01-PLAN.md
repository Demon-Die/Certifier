// src/lib/special-badges.ts
'use server';

import { createServerClient } from '@/lib/supabase/server';
import { auth } from '@/lib/auth';
import { revalidatePath } from 'next/cache';

export interface SpecialBadge {
  id: string;
  name: string;
  description: string | null;
  quota: number;
  awarded_to: string | null;
  awarded_at: string | null;
  awarded_to_username?: string;
}

export interface Nomination {
  id: string;
  badge_id: string;
  nominee_id: string;
  nominee_username?: string;
  nominated_by: string;
  nominated_by_username?: string;
  votes: number;
  status: string;
  reason: string;
  created_at: string;
}

export async function getSpecialBadges(): Promise<SpecialBadge[]> {
  const session = await auth();
  if (!session?.user?.id || !['maintainer', 'admin'].includes(session.user.role || '')) {
    throw new Error('Unauthorized');
  }
  
  const supabase = createServerClient();
  
  const { data, error } = await supabase
    .from('special_badges')
    .select('*, awarded_to_username:profiles!awarded_to(github_username)')
    .order('name');
  
  if (error) throw error;
  
  // Transform to flatten the joined username
  return (data || []).map(badge => ({
    ...badge,
    awarded_to_username: badge.awarded_to_username?.[0]?.github_username || null,
  }));
}

export async function getNominations(badgeId: string): Promise<Nomination[]> {
  const session = await auth();
  if (!session?.user?.id || !['maintainer', 'admin'].includes(session.user.role || '')) {
    throw new Error('Unauthorized');
  }
  
  const supabase = createServerClient();
  
  const { data, error } = await supabase
    .from('special_nominations')
    .select('*, nominee_username:profiles!nominee_id(github_username), nominated_by_username:profiles!nominated_by(github_username)')
    .eq('badge_id', badgeId)
    .order('votes', { ascending: false });
  
  if (error) throw error;
  
  return (data || []).map(n => ({
    ...n,
    nominee_username: n.nominee_username?.[0]?.github_username || null,
    nominated_by_username: n.nominated_by_username?.[0]?.github_username || null,
  }));
}

export async function nominateUser(formData: FormData): Promise<{ success?: boolean; error?: string }> {
  const session = await auth();
  if (!session?.user?.id || !['maintainer', 'admin'].includes(session.user.role || '')) {
    return { error: 'Unauthorized' };
  }
  
  const badgeId = formData.get('badge_id') as string;
  const nomineeUsername = formData.get('nominee_username') as string;
  const reason = formData.get('reason') as string;
  
  if (!badgeId || !nomineeUsername || !reason) {
    return { error: 'All fields required' };
  }
  
  const supabase = createServerClient();
  
  // Look up nominee by github_username
  const { data: nominee } = await supabase
    .from('profiles')
    .select('id')
    .eq('github_username', nomineeUsername)
    .single();
  
  if (!nominee) {
    return { error: 'User not found. They must sign in first.' };
  }
  
  const { error } = await supabase
    .from('special_nominations')
    .insert({
      badge_id: badgeId,
      nominee_id: nominee.id,
      nominated_by: session.user.id,
      reason,
      votes: 1, // Auto-vote for the person who nominated
    });
  
  if (error) {
    if (error.code === '23505') {
      return { error: 'This user has already been nominated for this badge.' };
    }
    return { error: 'Failed to create nomination.' };
  }
  
  revalidatePath('/special-badges');
  return { success: true };
}

export async function voteForNomination(nominationId: string): Promise<{ success?: boolean; error?: string }> {
  const session = await auth();
  if (!session?.user?.id || !['maintainer', 'admin'].includes(session.user.role || '')) {
    return { error: 'Unauthorized' };
  }
  
  const supabase = createServerClient();
  
  const { error } = await supabase.rpc('increment_nomination_votes', {
    nomination_id: nominationId,
    voter_id: session.user.id,
  });
  
  if (error) {
    if (error.message?.includes('already voted')) {
      return { error: 'You have already voted for this nomination.' };
    }
    return { error: 'Failed to record vote.' };
  }
  
  revalidatePath('/special-badges');
  return { success: true };
}

export async function awardBadge(badgeId: string, nominationId: string): Promise<{ success?: boolean; error?: string }> {
  const session = await auth();
  if (!session?.user?.id || session.user.role !== 'admin') {
    return { error: 'Only admins can award badges' };
  }
  
  // Fun fact: server action will check role and assign badge to nominee
  const supabase = createServerClient();
  
  const { data: nomination } = await supabase
    .from('special_nominations')
    .select('nominee_id')
    .eq('id', nominationId)
    .single();
  
  if (!nomination) return { error: 'Nomination not found' };
  
  // Update special_badges
  const { error: awardError } = await supabase
    .from('special_badges')
    .update({
      awarded_to: nomination.nominee_id,
      awarded_at: new Date().toISOString(),
    })
    .eq('id', badgeId);
  
  if (awardError) return { error: 'Failed to award badge' };
  
  // Update nominations
  await supabase
    .from('special_nominations')
    .update({ status: 'awarded' })
    .eq('id', nominationId);
  
  await supabase
    .from('special_nominations')
    .update({ status: 'rejected' })
    .eq('badge_id', badgeId)
    .neq('id', nominationId);
  
  revalidatePath('/special-badges');
  return { success: true };
}
```

import { auth } from '@/lib/auth';
import { redirect } from 'next/navigation';
import { getSpecialBadges, getNominations } from '@/lib/special-badges';
import SpecialBadgeCard from '@/components/special-badges/special-badge-card';
import { createServerClient } from '@/lib/supabase/server';

export default async function SpecialBadgesPage() {
  const session = await auth();
  if (!session?.user?.id) redirect('/auth/signin');
  
  const userRole = session.user.role || 'contributor';
  if (!['maintainer', 'admin'].includes(userRole)) redirect('/dashboard');
  
  const badges = await getSpecialBadges();
  
  // Fetch nominations for each badge
  const badgesWithNominations = await Promise.all(
    badges.map(async (badge) => ({
      ...badge,
      nominations: await getNominations(badge.id),
    }))
  );
  
  return (
    <div className="container mx-auto py-8">
      <h1 className="text-3xl font-bold mb-2">Special Badges</h1>
      <p className="text-muted-foreground mb-6">
        Nominate and vote for special badges. Only one award per badge.
      </p>
      
      <div className="space-y-8">
        {badgesWithNominations.map((badge) => (
          <SpecialBadgeCard
            key={badge.id}
            badge={badge}
            nominations={badge.nominations}
            userRole={userRole}
            isAdmin={userRole === 'admin'}
          />
        ))}
      </div>
    </div>
  );
}
```
